🖨️ Print SDK 示例

1. SDK 初始化（无需配置）

import { createPrintSDK } from '@jcyao/print-sdk';

// 创建 SDK 实例（无需配置，完全解耦）
const sdk = createPrintSDK();

2. 测试打印功能

3. 使用说明

// 直接打印（隐藏 iframe）
await sdk.print({
  template: templateData,  // 直接传入模板数据
  data: { /* 业务数据 */ },
  preview: false
});

// 预览后打印（新窗口）
await sdk.print({
  template: templateData,
  data: { /* 业务数据 */ },
  preview: true
});

// 仅生成 HTML（不打印）
const html = await sdk.generateHTML(template, data);

// 批量打印（同模板多数据）
await sdk.printMultiple(template, dataList, {
  preview: true,
  onProgress: (progress) => console.log(progress)
});
